
CREATE DATABASE ElibraryDB;

CREATE TABLE E_BOOK (
    CALLNO varchar(4000),
    NAME varchar(4000),
    AUTHOR varchar(4000),
	PUBLISHER varchar(4000),
	QUANTITY int,
	ISSUED int,
    CONSTRAINT "E_BOOK_PK" PRIMARY KEY ("CALLNO") ENABLE
);

CREATE TABLE E_LIBRARIAN (
    ID int,
    NAME varchar(4000),
    PASSWORD varchar(4000),
    EMAIL varchar(4000),
    MOBILE int,
	CONSTRAINT "E_LIBRARIAN_PK" PRIMARY KEY ("ID") ENABLE
);

CREATE TABLE E_ISSUEBOOK (
    CALLNO varchar(4000) NOT NULL ENABLE,
    STUDENTID varchar(4000) NOT NULL ENABLE,
    STUDENTNAME varchar(4000),
    STUDENTMOBILE int,
	ISSUEDDATE date,
    RETURNSTATUS varchar(4000)
);
